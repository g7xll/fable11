interface IERC721 {
    function ownerOf(uint256 tokenId) external view
        returns (address);
}

/// @title NFT Ownership Verifier
contract NFTOwnershipVerifier {
    function verifyOwnership(
        address nftContract,
        uint256 tokenId,
        address user
    ) external view returns (bool) {
        IERC721 nft = IERC721(nftContract);
        return nft.ownerOf(tokenId) == user;
    }
}
